    private void init() &#123;

        // 加载配置文件并初始化servletMap

        try &#123;
            String path = this.getClass().getResource("/").getPath();
            FileInputStream fis = new FileInputStream(path + "web.properties");
            webxml.load(fis);

            // 初始化 servletMap

            for (Object o : webxml.keySet()) &#123;
                String key = o.toString();
                if (key.endsWith(".url")) &#123;

                    String url = webxml.getProperty(key);
                    // className
                    String className = webxml.getProperty(servletName + ".className");
                    Servlet servlet = (Servlet) Class.forName(className).newInstance();
                    servletMap.put(url, servlet);
                &#125;
            &#125;
        &#125; catch (FileNotFoundException e) &#123;
            e.printStackTrace();
        &#125; catch (IOException e) &#123;

    public void start() &#123;

        init();
        // netty 封装了NIO, Reactor模型. BOss Worker,
        // Boss 线程
        EventLoopGroup bossGroup = new NioEventLoopGroup();
        // Worker 线程
        EventLoopGroup workerGroup = new NioEventLoopGroup();
        try &#123;


            ServerBootstrap bootstrap = new ServerBootstrap();
            //链路式编程
            bootstrap.group(bossGroup, workerGroup)
                    //主线程处理类， 底层使用反射
                    .channel(NioServerSocketChannel.class)

                    .childHandler(new ChannelInitializer<SocketChannel>() &#123;
                        @Override
                        protected void initChannel(SocketChannel socketChannel) throws Exception &#123;
                            // 无锁化串行编程
                            // Netty 对Http协议的封装,顺序有要求
// HttpResponseEncoder 编码器
                            socketChannel.pipeline().addLast(new HttpResponseEncoder());
                            //HttpRequestDecoder 解码器
                            socketChannel.pipeline().addLast(new HttpRequestDecoder());


                            //  业务处理类
                            socketChannel.pipeline().addLast(new TomcatHandler(servletMap));
                        &#125;
                    &#125;)
                    // 针对主线程的配置,分配线程最大数量  128

            System.out.println("Tomcat 启动成功, 端口号为: " + this.port);
            channelFuture.channel().closeFuture().sync();
